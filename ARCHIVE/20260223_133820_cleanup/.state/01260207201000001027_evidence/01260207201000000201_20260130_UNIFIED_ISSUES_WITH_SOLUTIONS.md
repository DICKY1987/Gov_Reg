# id_allocator.py
import datetime

class IDAllocator:
    def __init__(self, counter_file=".state/ID_COUNTER.json"):
        self.counter_file = counter_file
        self.load_counter()

    def allocate(self):
        timestamp = datetime.datetime.utcnow().strftime("%Y%m%d%H%M%S")
        sequence = str(self.counter).zfill(6)
        file_id = f"{timestamp}{sequence}"

        self.counter += 1
        self.save_counter()

        return file_id

    def load_counter(self):
        if os.path.exists(self.counter_file):
            data = json.load(open(self.counter_file))
            self.counter = data['last_sequence'] + 1
        else:
            self.counter = 0

    def save_counter(self):
        json.dump({
            "last_sequence": self.counter,
            "last_allocated": datetime.datetime.utcnow().isoformat()
        }, open(self.counter_file, 'w'))

# Usage

new_id = allocator.allocate()  # 20260130170728000042

import ast
import re

def extract_python_imports(file_path):
    """Extract import statements from Python file"""
    with open(file_path) as f:
        tree = ast.parse(f.read())

    imports = []
    for node in ast.walk(tree):
        if isinstance(node, ast.Import):
            for alias in node.names:
                imports.append(alias.name)
        elif isinstance(node, ast.ImportFrom):
            imports.append(node.module)

    return imports

def build_dependency_graph(registry):
    """Build depends_on relationships"""
    files_by_path = {f['relative_path']: f for f in registry['files']}

    for file_record in registry['files']:
        if file_record['extension'] == 'py':
            imports = extract_python_imports(file_record['relative_path'])

            depends_on = []
            for imp in imports:
                # Map import to file_id
                module_path = imp.replace('.', '/') + '.py'
                if module_path in files_by_path:
                    depends_on.append(files_by_path[module_path]['file_id'])

            # Find entry for this file
            entry = find_entry(registry, file_record['file_id'])

def detect_cycles(registry):
    """Detect cycles in DEPENDS_ON graph"""
    from collections import defaultdict, deque

    # Build adjacency list
    graph = defaultdict(list)
    for entry in registry['entries']:
        source = entry['file_id']
        for target in entry.get('depends_on_file_ids', []):
            graph[source].append(target)

    # DFS cycle detection
    def has_cycle(node, visited, rec_stack):
        visited.add(node)
        rec_stack.add(node)

        for neighbor in graph[node]:
            if neighbor not in visited:
                if has_cycle(neighbor, visited, rec_stack):
                    return True
            elif neighbor in rec_stack:
                return True

        rec_stack.remove(node)
        return False

    visited = set()
    rec_stack = set()

    for node in graph:
        if node not in visited:
            if has_cycle(node, visited, rec_stack):
                return True  # Cycle detected

    return False

# Execute

    raise ValidationError("Circular dependency detected in DEPENDS_ON graph")

    """Mark canonical vs duplicate files"""
    from collections import defaultdict

    # Group by embedded ID (files with same ID in filename)
    by_embedded_id = defaultdict(list)

    for file_record in registry['files']:
        path = file_record['relative_path']
        match = re.search(r'(\d{20})', path)
        if match:
