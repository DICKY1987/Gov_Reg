def compute_ast_hash(tree: ast.AST) -> str:
    """
    Deterministic hash ensures same code → same hash
    """
    # Stable dump: exclude line numbers/positions
    ast_dump = ast.dump(tree, include_attributes=False, indent=None)

    # SHA256 for cryptographic guarantee
    return hashlib.sha256(ast_dump.encode('utf-8')).hexdigest()

```python
def stable_topological_sort(headers: List[dict]) -> List[dict]:
    """
    Ensures same dependency graph → same execution order
    """
    # Build adjacency list from depends_on
    graph = {h['header']: h['depends_on'] for h in headers}

    # Kahn's algorithm with lexicographic tie-breaking
    sorted_headers = []
    while graph:
        # Select nodes with no dependencies, sorted alphabetically
        ready = sorted([k for k, v in graph.items() if not v])
        sorted_headers.extend(ready)
        graph = {k: [d for d in v if d not in ready]
                 for k, v in graph.items() if k not in ready}

    return sorted_headers
```
