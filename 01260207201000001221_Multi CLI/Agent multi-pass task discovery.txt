3. Task Deduplication Logic
Your Fingerprinting Approach:

// Pseudo-code from your design
function fingerprint(task) {
  return hash({
    type: task.type,
    targets: normalize(task.targets),  // Sort, canonicalize paths
    title: normalize(task.title)       // Lowercase, remove articles
  });
}

function deduplicate(tasklists) {
  const seen = new Map();
  const merged = [];
  
  for (const task of tasklists.flatMap(tl => tl.tasks)) {
    const fp = fingerprint(task);
    
    if (seen.has(fp)) {
      // Merge: union instructions, keep highest priority
      const existing = seen.get(fp);
      existing.instructions = [...new Set([...existing.instructions, ...task.instructions])];
      existing.acceptance_tests = [...new Set([...existing.acceptance_tests, ...task.acceptance_tests])];
      existing.priority = max(existing.priority, task.priority);
    } else {
      seen.set(fp, task);
      merged.push(task);
    }
  }
  
  return merged;
}


My plan had “regression detection” but not this proactive deduplication before execution. This is critical because:
	∙	Multiple passes will independently discover the same issue (e.g., “missing error handling in auth.py” found by both security and reliability passes)
	∙	Without deduplication, you’d execute redundant tasks
	∙	With merge logic, you get the union of insights from different perspectives

def validate_task_completeness(task):
    issues = []

    if not task.get("acceptance_tests"):

            "severity": "BLOCKING",
            "field": "acceptance_tests",
            "reason": "Task cannot be verified as complete without tests"
        })

            "severity": "BLOCKING", 
            "field": "targets",
            "reason": "Task must specify files/symbols it modifies"

    # Soft requirements (warnings)
    if not task.get("instructions") or len(task["instructions"]) == 0:
        issues.append({
            "severity": "WARNING",
            "field": "instructions",
            "reason": "Task should have explicit implementation steps"
        })
    
    if task.get("confidence", 1.0) < 0.5:
        issues.append({
            "severity": "WARNING",
            "field": "confidence",
            "reason": "Low confidence tasks should be reviewed before execution"
        })
    
    return issues

# In merge phase

def calculate_convergence_score(tasks, errors):
    # Weighted quality factors
    completeness = len([t for t in tasks if t.get("acceptance_tests")]) / len(tasks)
    testability = avg([rate_test_quality(t["acceptance_tests"]) for t in tasks])
    risk_assessed = len([t for t in tasks if t.get("risk_level")]) / len(tasks)
    
    # Error severity penalties
    critical_penalty = len([e for e in errors if e["severity"] == "CRITICAL"]) * 0.2
    high_penalty = len([e for e in errors if e["severity"] == "HIGH"]) * 0.1
    
    score = (0.4 * completeness + 0.3 * testability + 0.3 * risk_assessed) - critical_penalty - high_penalty
    return max(0.0, min(1.0, score))


This allows adaptive loop control:
	∙	If score improves by <0.05 per iteration → plateau detected → halt
	∙	If score regresses → rollback to previous version

#━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
echo "[PHASE 3] Starting iterative refinement..."

CURRENT_INPUT="$RUN_DIR/merged/TASKS.MERGED.raw.json"
ITERATION=1

while [ $ITERATION -le $MAX_ITERATIONS ]; do
  echo "  Iteration $ITERATION/$MAX_ITERATIONS"
  ITER_DIR="$RUN_DIR/iterations/iteration_$(printf '%03d' $ITERATION)"
  mkdir -p $ITER_DIR
  
  # Validate current state
  cat $CURRENT_INPUT | \
    claude -p --output-schema schemas/error_log.schema.json \
    --prompt-file prompts/refinement/validate_dag.md \
    "Validate this task DAG for completeness, conflicts, and quality" \
    > $ITER_DIR/ERROR_LOG_V${ITERATION}.json \
    2> $RUN_DIR/logs/refinement_iter_${ITERATION}.log
  
  # Calculate convergence score
  CONVERGENCE=$(jq '.convergence_metrics.score' $ITER_DIR/ERROR_LOG_V${ITERATION}.json)
  CRITICAL_ERRORS=$(jq '[.errors[] | select(.severity=="CRITICAL")] | length' $ITER_DIR/ERROR_LOG_V${ITERATION}.json)
  
  echo "    Convergence: $CONVERGENCE (threshold: $CONVERGENCE_THRESHOLD)"
  echo "    Critical errors: $CRITICAL_ERRORS"
  
  # Check convergence
  if (( $(echo "$CONVERGENCE >= $CONVERGENCE_THRESHOLD" | bc -l) )) && [ $CRITICAL_ERRORS -eq 0 ]; then
    echo "✓ CONVERGED at iteration $ITERATION"
    cp $CURRENT_INPUT $RUN_DIR/TASKS.FINAL.adj.json
    break
  fi
  
  # Refine
  cat $CURRENT_INPUT $ITER_DIR/ERROR_LOG_V${ITERATION}.json | \
    claude -p --output-schema schemas/dag.schema.json \
    --prompt-file prompts/refinement/improve_tests.md \
    "Fix all errors in ERROR_LOG. Maintain DAG structure." \
    > $ITER_DIR/TASKS.REFINED_V${ITERATION}.json \
    2>> $RUN_DIR/logs/refinement_iter_${ITERATION}.log
  
  CURRENT_INPUT="$ITER_DIR/TASKS.REFINED_V${ITERATION}.json"
  ITERATION=$((ITERATION + 1))
done

if [ $ITERATION -gt $MAX_ITERATIONS ]; then
  echo "⚠ WARNING: Max iterations reached without full convergence"
  echo "  Final convergence score: $CONVERGENCE"
  echo "  Manual review required: $RUN_DIR/iterations/iteration_$(printf '%03d' $MAX_ITERATIONS)/"
  exit 2
fi

#━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
# PHASE 4: FINALIZATION
#━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
echo "[PHASE 4] Finalizing execution plan..."

# Generate execution metadata
cat > $RUN_DIR/execution_plan.yaml <<EOF
run_id: $RUN_ID
status: ready_for_execution
convergence_score: $CONVERGENCE
iterations_required: $((ITERATION - 1))
total_tasks: $(jq '.nodes | length' $RUN_DIR/TASKS.FINAL.adj.json)
critical_path_length: $(jq '.metadata.critical_path_length' $RUN_DIR/TASKS.FINAL.adj.json)
estimated_duration: $(jq -r '.metadata.estimated_duration' $RUN_DIR/TASKS.FINAL.adj.json)
EOF

echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
echo "SUCCESS: Executable plan ready"
echo "  Location: $RUN_DIR/TASKS.FINAL.adj.json"
echo "  Metadata: $RUN_DIR/execution_plan.yaml"
echo "  Audit trail: $RUN_DIR/iterations/"
echo "━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━"
